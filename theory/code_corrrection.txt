Find the Mistake & Correct It


Question 1
from fastapi import FastAPI
app = FastAPI
@app.get("/")
def home():
    return {"message": "Hello"}

correct code
from fastapi import FastAPI
app = FastAPI()
@app.get("/")
def home():
    return {"message": "Hello"}


Question 2
@app.post("/users")
def create_user(user):
    return user

correct code
@app.post("/users")
def create_user(user: dict):
    return user

correct code
@app.get("/products/{id}")
def get_product(id: int):
    return {"id": id + 1}
